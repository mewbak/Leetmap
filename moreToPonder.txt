  int 
partition(int arr[], int start, int end) {
  int low = start;
  int high = end;
  int pivot = arr[start];@NOTE

  while(low < high) {
    while(arr[low] <= pivot && low <= end) {
      low ++;
    }
    while(arr[high] > pivot && high >= 0) {
      high --;
    }

    if(low < high) {
      int temp = arr[low];
      arr[low]  = arr[high];
      arr[high] = temp;
    }
  }
  arr[start] = arr[high]; @NOTE
  arr[high] = pivot;
  return high;
}
